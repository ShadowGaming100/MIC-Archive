print("This code calculates the area of triangle with keyboard input.")

# Prompt the user to enter/input the base of the triangle.
base = int(input("Enter/Input the base of the triangle: "))

# Prompt the user to enter/input the height of the triangle.
height = int(input("Enter/Input the height of the triangle: "))

# Calculate the area of the triangle.
area = 0.5 * base * height

# Print the area of the triangle.
print("The area of triangle is", area)
